#include<iostream>

using namespace std;

// Declare function show with s is input
void show(string s) {
    // Convert s's uppercase letters to lowercase letters
    for (int i = 0; i < s.size(); i++) {
        if (s[i] >= 'A' && s[i] <= 'Z') {
            s[i] += 32;
        }
    }
    // Print converted string on the screen
    cout << s << endl;
}

int main() {
    string s1;
    cin >> s1;
    show(s1);
    return 0;
}
